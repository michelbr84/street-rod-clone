// src/components/HUD.tsx
 
import styles from '../styles/hud.module.css';
import { useRace } from '../context/RaceContext'; // üöó Importa o contexto da corrida
 
interface HUDProps {
  speed?: number;
  gear?: number;
  time?: number;
  fuel?: number;
  position?: number;
}
 
export default function HUD({ speed = 0, gear = 0, time = 0, fuel = 100, position = 5 }: HUDProps) {
  const { tire } = useRace(); // üõû Obt√©m informa√ß√µes dos pneus do contexto
 
  // ‚úÖ Verifica√ß√£o para evitar NaN e valores indefinidos
  const safeSpeed = isNaN(speed) ? 0 : speed;
  const safeGear = isNaN(gear) ? 0 : gear;
  const safeTime = isNaN(time) ? 0 : time;
  const safeFuel = isNaN(fuel) ? 0 : fuel;
  const safeTireDurability =
    tire?.durability !== undefined && !isNaN(tire.durability) ? tire.durability : 100;
 
  // üî• Feedback visual para o n√≠vel de combust√≠vel (verde, amarelo, vermelho)
  const fuelLevelClass =
    safeFuel > 50 ? styles.fuelHigh : safeFuel > 20 ? styles.fuelMedium : styles.fuelLow;
 
  // üö¶ Feedback visual para a velocidade (verde, laranja, vermelho)
  const speedStatus =
    safeSpeed > 180 ? styles.speedHigh : safeSpeed > 100 ? styles.speedMedium : styles.speedLow;
 
  // üõû Feedback visual para o desgaste dos pneus
  const tireWearClass =
    safeTireDurability > 70
      ? styles.tireGood
      : safeTireDurability > 40
        ? styles.tireWorn
        : styles.tireCritical;
 
  return (
    <div className={styles.hudContainer}>
      {/* üöÄ Velocidade */}
      <div className={`${styles.hudItem} ${speedStatus}`}>
        <span className={styles.icon}>üöÄ</span> Speed: <strong>{safeSpeed.toFixed(0)}</strong> km/h
      </div>
 
      {/* ‚öôÔ∏è Marcha */}
      <div className={styles.hudItem}>
        <span className={styles.icon}>‚öôÔ∏è</span> Gear:{' '}
        <strong>{safeSpeed === 0 ? 'N' : safeGear}</strong>
      </div>
 
      {/* ‚è±Ô∏è Tempo de corrida */}
      <div className={styles.hudItem}>
        <span className={styles.icon}>‚è±Ô∏è</span> Time: <strong>{safeTime.toFixed(1)}</strong> s
      </div>
 
      {/* ‚õΩ N√≠vel de combust√≠vel */}
      <div className={`${styles.hudItem} ${fuelLevelClass}`}>
        <span className={styles.icon}>‚õΩ</span> Fuel: <strong>{safeFuel.toFixed(1)}</strong>%
      </div>
 
      {/* üèÅ Posi√ß√£o atual na corrida */}
      <div className={styles.hudItem}>
        <span className={styles.icon}>üèÅ</span> Position: <strong>{position}</strong>/5
      </div>
 
      {/* üõû Estado do desgaste dos pneus */}
      {tire && (
        <div className={`${styles.hudItem} ${tireWearClass}`}>
          <span className={styles.icon}>üõû</span> Tire Wear:{' '}
          <strong>{safeTireDurability.toFixed(0)}</strong>%
        </div>
      )}
    </div>
  );
}
